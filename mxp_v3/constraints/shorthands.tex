\[
	\left\{ \begin{array}{lcl}
		\locPerspectiveSum _{i}
		& \define &
		\left[ \begin{array}{cl}
			+ & \isMxpInstructionDecoder _{i} \\
			+ & \isMxpMacro              _{i} \\
			+ & \isMxpScenario           _{i} \\
			+ & \isMxpComputation        _{i} \\
		\end{array} \right] \vspace{2mm} \\
		\locWeightSum _{i}
		& \define &
		\left[ \begin{array}{crcl}
