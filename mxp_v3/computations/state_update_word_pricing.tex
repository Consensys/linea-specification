\begin{description}

        \[
            \eucCall {

                argOne    = \mxpMacroSizeOneLo _{i - \locScenarioToMacroRowOffset} ,
                argTwo    = \evmWordSize                                           ,

        \]

        we impose that
        \[
            \left\{ \begin{array}{lcl}

        we impose
        \[

\end{description}
