\begin{enumerate}
    \item \userTransactionNumber{}:
	transaction number column, starts at 0 and is incremented by one when starting a new transaction.
	\item $\absLogNum$: number of the current log entry, only used in phase $\phaseLogEntries$;
    \item $\absLogNumMax$: number of log entries in a given transaction;
    \item $\isTxn$:
	bit column that defines the transaction perspective;
    \item $\isCmp$:
	bit column that defines the computation perspective;
    \item \col{LIMB\_CONSTRUCTED}:
	binary column;
	is equal to $1$ when the $\rlpTxnComputationColumnLimb$ is part of the RLP of the transaction receipt, otherwise is equal to $0$;
	abbreviated to $\lc$;
    \item $\index$:
	index of the current limb within the current string;
	starts at $0$ when starting a new transaction and increments by 1 when $\lc$ is one; 
    \item $\byteSizeCountdown$:
	used in \phaseRlpPrefix{} to compute the global \rlp{} prefix;
	in futher phases this column simply decreases by $\rlpTxnComputationColumnLimbSize$ when $\lc$ is one;
	constrained to vanish at the end of the transaction;
\end{enumerate}

\begin{enumerate}
	\setcounter{enumi}{27}
    \item $\phaseEnd$:
	binary column; equals $1$ on the last row of any phase, $0$ elsewhere;
    \item \ct{}:
	counter column; 
	either hovers around $0$ or counts continuously up from $0$ to \maxCt{} whereupon it resets;
    \item \maxCt:
	counter-constant column;
	indicates the value at which \ct{} must reset;
    \item $\done$:
	binary column; equal to $1$ on the final row of every \ct{} cycle; equal to $0$ otherwise;
\end{enumerate}
