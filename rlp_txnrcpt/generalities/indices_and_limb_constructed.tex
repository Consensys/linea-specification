\noindent We further impose:
\begin{enumerate}
    \item \If $\locIsFirstRowOfTransaction _{i} = 1$ \Then:
        \begin{enumerate}
            \item $\index _{i} = 0$
        \end{enumerate}
    \item \If $\locIsFirstRowOfTransaction _{i} = 0$ \Then:
        \begin{enumerate}
            \item $\index _{i} = \index _{i - 1} + \lc _{i - 1}$
        \end{enumerate}
        \saNote{}
        In other words, $\index$ resets to $0$ at the begining of a new transaction,
        and increases by $1$ following any row where $\lc$ lights up.
\end{enumerate}
