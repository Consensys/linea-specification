\begin{description}
    \item[\rlpUtilsInstDataPricing{}:]
        takes as inputs
        \begin{itemize}
            \item a right zero-padded limb \locLimb{};
            \item an \textbf{nonzero} integer \loc{nBytes} representing the number of significant bytes in that limb \loc{nBytes};
        \end{itemize}
        it returns a pair of small integers
        \begin{itemize}
            \item \loc{num\_zero\_bytes}
            \item \loc{num\_nonzero\_bytes}
            \item \locFirstByte{} the most ignificant bytes of the \locLimb
        \end{itemize}
        representing respectively the number of zero and nonzero bytes in the relevant bytes of the limb;
\end{description}
