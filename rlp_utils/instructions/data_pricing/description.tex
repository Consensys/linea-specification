\begin{description}
    \item[\rlpUtilsInstDataPricing{}:]
        takes as inputs
        \begin{itemize}
            \item a right zero-padded limb \locInputLimb{};
            \item an \textbf{nonzero} integer \locInputNBytes{} representing the number of significant bytes in that limb \locInputNBytes{};
        \end{itemize}
        it returns a pair of small integers
        \begin{itemize}
            \item \locOutputNumberOfZeroBytes{}
            \item \locOutputNumberOfNonzeroBytes{}
            \item \locOutputFirstByte{} the most ignificant bytes of the \locInputLimb{}
        \end{itemize}
        representing respectively the number of zero and nonzero bytes in the relevant bytes of the limb;
\end{description}
