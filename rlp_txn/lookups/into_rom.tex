\begin{description}
	\item[\underline{Selector:}] $\CFI_{i} \cdot \phaseData_{i} \cdot (1 - \isPrefix_{i}) \cdot \lc_{i} \neq 0$;
	\item[\underline{Source columns:}] from the \rlpTxnMod{} module:
	\begin{multicols}{3}
	\begin{enumerate}
		\item $\cfi_{i}$
		\item $\limb_{i}$
		\item $\indexData_{i}$
		\item $\nBytes_{i}$
	\end{enumerate}
	\end{multicols}
\item[\underline{Target columns:}] from the \romMod{} module: 
	\begin{multicols}{3}
	\begin{enumerate}
		\item $\cfi_{j}$
		\item $\limb_{j}$
		\item $\index_{j}$
		\item $\nBytes_{j}$
	\end{enumerate} 
	\end{multicols}
\end{description}
