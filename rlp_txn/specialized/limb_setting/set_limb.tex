We define \setLimbName{} as follows:
\[
	\begin{array}{l}
		\setLimb{
			anchorRow  = i                       ,
			relOffset  = \relof                  ,
			limb       = \loc{limb}              ,
			nBytes     = \loc{number\_of\_bytes} ,
		}
		\vspace{2mm} \\
		\qquad \define
		\left\{ \begin{array}{lcl}
			\lc       _{i + \relof} & = & \true                   \\
			\rlpTxnComputationColumnLimb     _{i + \relof} & = & \loc{limb}              \\
			\rlpTxnComputationColumnLimbSize _{i + \relof} & = & \loc{number\_of\_bytes} \\
		\end{array} \right.
	\end{array}
\]
