\begin{center}
\boxed{\text{In all this section, it is assumed that $\phaseRlpPrefix_{i}=1$.}}
\end{center}

We impose some constancies:
\begin{enumerate}
	\item $\rlptsize_{i} = \rlptsize_{i+1}$
	\item $\rlpxsize_{i} = \rlpxsize_{i+1}$
\end{enumerate}

We first determine the $\transactionType$ prefix concatenation on the first row:
\begin{enumerate}[resume]
	\item \If $\isTxn_{i-1}=0$ \Then
	\begin{enumerate}
		\item $\maxCt_{i}=0$
		\item $\lt_{i}=1$
		\item $\lx_{i}=1$
		\item $\phasend_{i}=0$
		\item \If $\typeZeroTx_{i}        = 0$ \Then $\lc_{i}=0$
		\item \If $\typeZeroTx_{i}_{i} \neq 0$ \Then:
		\begin{enumerate}
			\item $\lc_{i}      = 1 $
		 	\item $\limb_{i}    = \txType_{i-1} \cdot 256^{\llargeMO}$ 
		 	\item $\limbsize_{i}= 1 $
		\end{enumerate}
	\end{enumerate}
\end{enumerate}
We now compute the RLP prefix of $L_{\mathrm{T}}(T)$:
\begin{enumerate}[resume]
	\item \If $\isTxn_{i-2}=1$ \Then
	\begin{enumerate}
		\item $\lt_{i}=1$
		\item $\lx_{i}=0$
		\item $\inputColumn{1}_{i} = \rlptsize_{i}$
		\item $\inputColumn{3}_{i} = 1$ 
		\item 
				\[
    			\rlpBytestringPrefix_{i}
    			\left(
				\begin{array}{r}
    			\inputColumn{1},
    			\inputColumn{2},
    			\inputColumn{3},
    			\ct,
    			\maxCt; \\
    			\rlpUtilsFlag,
    			\INST,
    			\resultColumn{1},
    			\resultColumn{2},
    			\resultColumn{5}; \\
    			\limb,
    			\lc,
    			\limbsize; \\
    			\end{array}
				\right)
				\]
		\item \If $\phasend_{i}=0$
	\end{enumerate}
\end{enumerate}

\begin{enumerate}[resume]
	\item \If $\isTxn_{i-3}=1$ \Then
	\begin{enumerate}
		\item $\lt_{i}=0$
		\item $\lx_{i}=1$
		\item $\inputColumn{1}_{i} = \rlpxsize_{i}$
		\item $\inputColumn{3}_{i} = 1$ 
		\item 
				\[
    			\rlpBytestringPrefix_{i}
    			\left(

    			\inputColumn{1},

    			\inputColumn{3},

    			\limb,
    			\lc,

    			\end{array}
				\right)

	\end{enumerate}
\end{enumerate}
