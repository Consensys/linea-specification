We introduce the following simple constraint systems:
\[
	\left\{ \begin{array}{lcl}
		\limbBelongsToLtOnly {
			anchorRow = i      ,
			relOffset = \relof ,
		}
		& \define &
		\left\{ \begin{array}{lcl}
			\lt _{i + \relof} & = & \yellowm{1} \\
			\lx _{i + \relof} & = & 0 \\
		\end{array} \right.
		\vspace{2mm}
		\\
		\limbBelongsToLxOnly {
			anchorRow = i      ,
			relOffset = \relof ,
		}
		& \define &
		\left\{ \begin{array}{lcl}
			\lt _{i + \relof} & = & 0 \\
			\lx _{i + \relof} & = & \yellowm{1} \\
		\end{array} \right.
		\vspace{2mm}
		\\
		\limbBelongsToBothLtAndLx {
			anchorRow = i      ,
			relOffset = \relof ,
		}
		& \define &
		\left\{ \begin{array}{lcl}
			\lt _{i + \relof} & = & \yellowm{1} \\
			\lx _{i + \relof} & = & \yellowm{1} \\
		\end{array} \right.
		\\
	\end{array} \right.
\]
