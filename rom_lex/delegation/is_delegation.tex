We specify \byteCodeActuallyDelegationCode{} as follows:
\begin{enumerate}
	\item \If $\leadingThreeBytes _{i} \neq \delegationPrefix$ \Then $\byteCodeActuallyDelegationCode _{i} = \false$
	\item \If $\leadingThreeBytes _{i} =    \delegationPrefix$ \Then $\byteCodeActuallyDelegationCode _{i} = \true $
\end{enumerate}
We further impose the following:
\begin{enumerate}[resume]
	\item \If $\byteCodeActuallyDelegationCode _{i} = 0$ \Then
		\[
			\left\{ \begin{array}{lclc}
				\delegationAddressHi _{i} & = & 0 & (\sanityCheck) \\
				\delegationAddressLo _{i} & = & 0 & (\sanityCheck) \\
			\end{array} \right.
		\]
\end{enumerate}
