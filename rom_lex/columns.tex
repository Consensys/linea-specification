\begin{enumerate}
	\item \cfi{}: code fragment index;
	\item $\cfiMax{}$: conflation-wide maximal code fragment index;
	\item $\codeSize$: containing the code size of the bytecode;
	\item $\addrHi{}$ and $\addrLo{}$: high and low part of an address;
	\item \depNum{}: deployment number;
	\item \depStatus{}: deployment status;
	\item \delegationNumber{}: delegation number;
\end{enumerate}
The next batch of columns pertains to recognizing delegated accounts.
\begin{enumerate}[resume]
	\item \byteCodeCouldBeDelegationCode{}: binary column; indicates whether byte code \textbf{could be} delegation code and should therefore be investigated as such;
	\item \byteCodeActuallyDelegationCode{}: binary column; indicates whether byte code \textbf{actually is} delegation code;
	\item \leadingThreeBytes{}, \leadBytesOfDelegationAddress{} and \tailBytesOfDelegationAddress{}:
		contain, respectively the first 3, the next 4 and next 16 bytes of an \textbf{eligible} byte code fragment;
	\item \delegationAddressHi{} and \delegationAddressLo{}: hi and lo parts of a delegation address;
	\item $\CH\high$ and $\CH\low$ high and low parts of the code hash;
	\item $[\col{@}_{k}]$, $k = 1, \dots, \numberOfAtColumnsInRomLex$: binary columns;
	\item $\byteCol{k}$, $k\in\{0, 1,\dots,\llargeMO\}$: byte columns;
\end{enumerate}
