In the present section we perform the following checks that are \textbf{required} to pass.
\begin{description}
	\item[\underline{Ensuring the \col{auth.chain\_id} is well formed:}] 
		we enforce $\col{auth.chain\_id} < 2^{256}$
	\item[\underline{\cite{EIP-2681}-bound \col{auth.nonce} check:}]
		we enforce $\col{auth.nonce} < 2^{64}$
	\item[\underline{Ensuring the \col{auth.address} is well formed:}] 
		we enforce $\col{auth.address} < 2^{160}$
	\item[\underline{Ensuring the \col{auth.y\_parity} is well formed:}] 
		we enforce $\col{auth.y\_parity} < 2^{8}$
	\item[\underline{Ensuring the \col{auth.r} is well formed:}] 
		we enforce $\col{auth.r} < 2^{256}$
	\item[\underline{Ensuring the \col{auth.s} is well formed:}] 
		we enforce $\col{auth.s} < 2^{256}$
\end{description}
