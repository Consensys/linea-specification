\begin{center}
	\boxed{\text{%
        The following constraints assume that
	$\left\{ \begin{array}{lcl}
		\isCoinbase _{i - 1} & = & 0 \\
		\isCoinbase _{i}     & = & 1 \\
	\end{array} \right.$}}
\end{center}
\begin{description}
	\item[\underline{\underline{Setting \inst{COINBASE}:}}]
		we impose
		\[
			\left\{ \begin{array}{lcl}
				\currDataHi & = & \coinbase\high _{i} \\
				\currDataLo & = & \coinbase\low  _{i} \\
			\end{array} \right.
		\]
	\item[\underline{\underline{\inst{GASLIMIT} deviation lower bound:}}]
		\def\rowOffset{\yellowm{0}}
		we impose
		\[
			\wcpCallToLt{
				anchorRow = i           ,
				relOffset = \rowOffset  ,
				argOneHi  = \currDataHi ,
				argOneLo  = \currDataLo ,
				argTwoHi  = 256^\ssmall ,
				argTwoLo  = 0           ,
			}
		\]
		\saNote{}
		The above ensures that \inst{COINBASE} fits in a $\addressSize$-byte integer.
\end{description}
