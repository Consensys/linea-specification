The lookup is structured as follows:
\begin{description}
	\item[\underline{Selector:}] we use $\col{wcp\_selector} \define \wcpFlag _{i}$
	\item[\underline{Source columns:}] ---
		\begin{multicols}{3}
			\begin{enumerate}
				\item $\argOneHi       _{i}$
				\item $\argOneLo       _{i}$
				\item $\argTwoHi       _{i}$
				\item $\argTwoLo       _{i}$
				\item $\res            _{i}$
				\item $\exoInstruction _{i}$
			\end{enumerate}
		\end{multicols}
	\item[\underline{Target columns:}] ---
		\begin{multicols}{3}
			\begin{enumerate}
				\item $\argOneHi _{j}$
				\item $\argOneLo _{j}$
				\item $\argTwoHi _{j}$
				\item $\argTwoLo _{j}$
				\item $\res      _{j}$
				\item $\INST     _{j}$
			\end{enumerate}
		\end{multicols}
\end{description}
