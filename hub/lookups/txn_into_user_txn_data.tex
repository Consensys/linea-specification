The present section describes the lookup between the \hubMod{} and the \userTxnDataMod{} module.
\begin{description}
	\item[Source selector:]
		we use the following source selector: $\col{src\_sel} _{i} \define \peekTransaction _{i}$
	\item[Target selector:]
		we use the following target selector: $\col{tgt\_sel} _{j} \define \isUserTxnHubView _{j}$
		\item[\underline{Column correspondence:}]
			we impose the following correspondence
			\begingroup
			\renewcommand{\arraystretch}{1.2}
			\[
				\resizebox{.95\linewidth}{!}{$
				\begin{array}{|l|c|l|}
				\hline
				\text{\hubMod{}-columns}     &                  & \text{\txnDataMod{}-columns} \\ \hline\hline
				% numbers
				\totalTransactionNumber _{i} & \rightsquigarrow & \totalTransactionNumber _{j} \\ \hline
				\blockNumber            _{i} & \rightsquigarrow & \blockNumber            _{j} \\ \hline
				% addresses
				\user _{i} \cdot \txFrom\high     _{i} & \rightsquigarrow & \user _{j} \cdot \txnDataHubFromHi     _{j} \\ \hline
				\user _{i} \cdot \txFrom\low      _{i} & \rightsquigarrow & \user _{j} \cdot \txnDataHubFromLo     _{j} \\ \hline

				\user _{i} \cdot \txCoinbase\high _{i} & \rightsquigarrow & \user _{j} \cdot \txnDataHubCoinbaseHi _{j} \\ \hline

	\end{description}
