\item[\underline{\underline{Delegate (or recipient) account-row n$^°~\bm{(i + \roffTxSkipUserRecipientOrDelegationAccount)}$:}}]
	this account-row was included in order to display the delegated account,
	in case of a \locIsMessageCallTransaction{} transaction where the target account is delegated;
	in all other cases the present account row simply reproduces the target account.
	We define the following shorthand
	\[
		\left\{ \begin{array}{lcl}
			\locLoadDelegateAccount & \define & \locIsMessageCallTransaction \cdot \locRecipientIsDelegated \\
			\locLoadRecipientAccountAgain  & \define &
			\left[ \begin{array}{clcl}
				+ & \locIsMessageCallTransaction  & \cdot & \locRecipientIsntDelegated \\
				+ & \locIsDeploymentTransaction  \\
			\end{array} \right]
			\\
		\end{array} \right.
	\]
	We now impose constraints depending on which of the above is the case.
	\begin{description}
		\item[\underline{The ``load delegation'' case:}] 
			\If   $\locLoadDelegateAccount = 1$
			\Then we impose
			\[
				\left\{ \begin{array}{lcl}
					\accAddress \high         _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & \accDelegationAddressHi  _{i + \roffTxSkipUserRecipientAccount}              \\
					\accAddress \low          _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & \accDelegationAddressLo  _{i + \roffTxSkipUserRecipientAccount} \vspace{2mm} \\
					\multicolumn{3}{l}{\accCheckForDelegationIfAccountHasCode { anchorRow = i, relOffset = \roffTxSkipUserRecipientOrDelegationAccount } } \\
					% \accCheckForDelegation    _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & \accHasCode              _{i + \roffTxSkipUserRecipientOrDelegationAccount}  \\
					% \accCheckForDelegationNew _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & 0 \quad (\trash)                                                             \\
				\end{array} \right.
			\]
		\item[\underline{The ``load to account'' case:}]
			\If   $\locLoadRecipientAccountAgain = 1$
			\Then we impose
			\[
				\left\{ \begin{array}{l}
					\accSameAddr  {i}{\roffTxSkipUserRecipientOrDelegationAccount}{\roffTxSkipUserRecipientAccount} \vspace{2mm} \\
					% \accAddress \high _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & \txTo  \high  _{i + \roffTxSkipUserTransaction}              \\
					% \accAddress \low  _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & \txTo  \low   _{i + \roffTxSkipUserTransaction} \vspace{2mm} \\
					\accDontCheckForDelegation { anchorRow = i, relOffset = \roffTxSkipUserRecipientOrDelegationAccount } \\
					% \accCheckForDelegation    _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & 0 \quad (\trash) \\
					% \accCheckForDelegationNew _{i + \roffTxSkipUserRecipientOrDelegationAccount} & = & 0 \quad (\trash) \\
				\end{array} \right.
			\]
		\item[\underline{The ``common'' account operations:}]
			In either case we impose
			\[
				\left\{ \begin{array}{lcl}
					\accSameBalance                        {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\accSameNonce                          {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\accSameCode                           {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\accSameDeployment                     {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\accCheckForDelegationIfAccountHasCode { anchorRow = i, relOffset = \roffTxSkipUserRecipientOrDelegationAccount} \\
					% \accConditionallyCheckForDelegation { anchorRow = i, relOffset = \roffTxSkipUserRecipientOrDelegationAccount, condition = \locCheckDelegateForDelegation } \\
					\accSameWarmth                         {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\accSameMarkedForDeletionFlag          {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\accIsntPrecompile                     {i}{\roffTxSkipUserRecipientOrDelegationAccount} \\
					\standardDomSubStamps {
						anchorRow = i                                           ,
						relOffset = \roffTxSkipUserRecipientOrDelegationAccount ,
						domOffset = \roffTxSkipUserRecipientOrDelegationAccount ,
					}
					\\
				\end{array} \right.
			\]
			% where we have set
			% \[
			% 	\locCheckDelegateForDelegation

			% 	\left[ \begin{array}{cl}
			% 		\cdot & \locLoadDelegateAccount     \\
			% 		\cdot & \locDelegateOrCalleeHasCode \\
			% 	\end{array} \right]

	\end{description}
