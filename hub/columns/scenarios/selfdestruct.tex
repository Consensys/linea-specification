\begin{enumerate}[resume]
	\item \scenSelfdestructException:
		binary column; lights up whenever the \zkEvm{} is about to attempt to run a \inst{SELFDESTRUCT} instruction and it raises an exception;
	\item \scenSelfdestructWillRevert:
		binary column; lights up whenever the \zkEvm{} runs an unexceptional \inst{SELFDESTRUCT} instruction which will get reverted;
	\item \scenSelfdestructWontRevertAlreadyMarked:
		binary column; lights up whenever the \zkEvm{} runs an unexceptional \inst{SELFDESTRUCT} instruction which won't get reverted on an account already $\accMarkedForSelfdestruct$;
	\item \scenSelfdestructWontRevertNotYetMarked:
		binary column; lights up whenever the \zkEvm{} runs an unexceptional \inst{SELFDESTRUCT} instruction which won't get reverted on an account not yet $\accMarkedForSelfdestruct$;
\end{enumerate}
