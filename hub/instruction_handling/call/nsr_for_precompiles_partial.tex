\begin{center}
	\boxed{%
		\text{The constraints presented below are written under the assumption that } 
			\left\{ \begin{array}{lcl}
				\peekScenario _{i} & = & 1 \\
				\scenCallPrc  _{i} & = & 1 \\
			\end{array} \right.
			}
\end{center}
\begin{description}
	\item[\underline{Setting the first half of peeking flags:}]
		we use the following shorthand
		\[
			\nsrPrecompilesFH
			\define
			\left[ \begin{array}{lrcl}
				+ & (\callPrcFailureSecondScenarioRowOffset           + 1) & \!\!\! \cdot \!\!\! & \scenCallPrcFailure           _{i + \callFirstScenarioRowOffset } \\
				+ & (\callPrcSuccessWillRevertSecondScenarioRowOffset + 1) & \!\!\! \cdot \!\!\! & \scenCallPrcSuccessWillRevert _{i + \callFirstScenarioRowOffset } \\
				+ & (\callPrcSuccessWontRevertSecondScenarioRowOffset + 1) & \!\!\! \cdot \!\!\! & \scenCallPrcSuccessWontRevert _{i + \callFirstScenarioRowOffset } \\
			\end{array} \right]
		\]
		and use it to set the first few non stack-rows of \inst{CALL}'s to precompiles:
		\[
			\hspace*{-2cm}
			\left[ \begin{array}{lrcl}
				+ \!\!\! & \peekingSumPrcFailureFH           & \!\!\! \cdot \!\!\! & \scenCallPrcFailure           _{i + \callFirstScenarioRowOffset } \\
				+ \!\!\! & \peekingSumPrcSuccessWillRevertFH & \!\!\! \cdot \!\!\! & \scenCallPrcSuccessWillRevert _{i + \callFirstScenarioRowOffset } \\
				+ \!\!\! & \peekingSumPrcSuccessWontRevertFH & \!\!\! \cdot \!\!\! & \scenCallPrcSuccessWontRevert _{i + \callFirstScenarioRowOffset } \\
			\end{array} \right]
			=
			\nsrPrecompilesFH
		\]
	\item[\underline{Propagating the current \hubStamp:}]
		we impose that the current \hubStamp{} remain valid at least for a while:
		\[
			\left[ \begin{array}{lrcl}
				+ \!\!\! & \Big[ \hubStamp _{i + \callPrcFailureSecondScenarioRowOffset           } - \hubStamp _{i - \callFirstStackRowOffset } \Big] & \!\!\! \cdot \!\!\! & \scenCallPrcFailure           _{i + \callFirstScenarioRowOffset } \vspace{2mm} \\
				+ \!\!\! & \Big[ \hubStamp _{i + \callPrcSuccessWillRevertSecondScenarioRowOffset } - \hubStamp _{i - \callFirstStackRowOffset } \Big] & \!\!\! \cdot \!\!\! & \scenCallPrcSuccessWillRevert _{i + \callFirstScenarioRowOffset } \vspace{2mm} \\
				+ \!\!\! & \Big[ \hubStamp _{i + \callPrcSuccessWontRevertSecondScenarioRowOffset } - \hubStamp _{i - \callFirstStackRowOffset } \Big] & \!\!\! \cdot \!\!\! & \scenCallPrcSuccessWontRevert _{i + \callFirstScenarioRowOffset } \\
			\end{array} \right] = 0
		\]
\end{description}
