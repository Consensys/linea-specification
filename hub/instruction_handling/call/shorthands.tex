We further define \locCalleeIsDelegatedToSelf{} as follows:
\begin{enumerate}
        \item \If $\locCalleeIsDelegated = \false $ \Then $\locCalleeIsDelegatedToSelf \define \false$
        \item \If $\locCalleeIsDelegated = \true  $ \Then
		\begin{enumerate}
		        \item
				\If   $\locCalleeDelegateAddressHi \neq \locCalleeAddressHi $ \Then $\locCalleeIsDelegatedToSelf \define \false$
		        \item                                                               
				\If   $\locCalleeDelegateAddressLo \neq \locCalleeAddressLo $ \Then $\locCalleeIsDelegatedToSelf \define \false$
		        \item                                                               
				\If   $\locCalleeDelegateAddressHi =    \locCalleeAddressHi $
				\et   $\locCalleeDelegateAddressLo =    \locCalleeAddressLo $ \Then $\locCalleeIsDelegatedToSelf \define \true$
		\end{enumerate}
\end{enumerate}
