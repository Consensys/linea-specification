We further define \locCalleeIsDelegatedToSelf{} as follows:
\begin{enumerate}
	\item \If $\locCalleeIsDelegated = \false $ \Then $\locCalleeIsDelegatedToSelf \define \false$
	\item \If $\locCalleeIsDelegated = \true  $ \Then
		\begin{enumerate}
				\item
					\If   $\locCalleeDelegateAddressHi \neq \locCalleeAddressHi $ \Then
					\[
						\locCalleeIsDelegatedToSelf \define \false
					\]
				\item
					\If   $\locCalleeDelegateAddressLo \neq \locCalleeAddressLo $ \Then
					\[
						\locCalleeIsDelegatedToSelf \define \false
					\]
			\item
				\If
				\[
					\left\{ \begin{array}{lclr}
						\locCalleeDelegateAddressHi & = & \locCalleeAddressHi  & \et \\
						\locCalleeDelegateAddressLo & = & \locCalleeAddressLo \\
					\end{array} \right.
				\]
				\Then $\locCalleeIsDelegatedToSelf \define \true$
		\end{enumerate}
\end{enumerate}
