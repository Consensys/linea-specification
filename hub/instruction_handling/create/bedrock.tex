We impose the following constraints.
\begin{description}
	\item[\underline{Forward setting of \emph{some} \inst{CREATE}-scenario:}] we enforce that
		\[
			\If
			\left\{ \begin{array}{lcl}
				\peekStack          _{i} & = & 1 \\
				\stackDecCreateFlag _{i} & = & 1 \\
				\tliCounter         _{i} & = & 0 \\
				\stackSux           _{i} & = & 0 \\
			\end{array} \right.
			\Then
			\left\{ \begin{array}{lcl}
				\peekScenario  _{i + 2} & = & 1 \\
				\scenCreateSum _{i + 2} & = & 1 \\
			\end{array} \right.
		\]
\end{description}
