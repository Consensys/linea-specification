	\boxed{\text{All constraints in this subsection assume that $\peekAccount _{i} = 1$.}}

We specify the behaviour of the $\accIsDelegated$ and $\accIsDelegatedNew$ columns
which indicate whether the account is delegated as per \cite{EIP-7702}.
\begin{enumerate}
	\item \If $\accDelegationAddressHi _{i} \neq 0$ \Then $\accIsDelegated _{i} = 1$
	\item \If $\accDelegationAddressHi _{i} =    0$ \Then
		\begin{enumerate}
			\item \If $\accDelegationAddressLo _{i} \neq 0$ \Then $\accIsDelegated _{i} = 1$
			\item \If $\accDelegationAddressLo _{i} =    0$ \Then $\accIsDelegated _{i} = 0$
		\end{enumerate}
\end{enumerate}
We enforce the analogous constraints for $\accHasCode\new$:
\begin{enumerate}[resume]
	\item \If $\accDelegationAddressNewHi _{i} \neq 0$ \Then $\accIsDelegatedNew _{i} = 1$
	\item \If $\accDelegationAddressNewHi _{i} =    0$ \Then
		\begin{enumerate}
			\item \If $\accDelegationAddressNewLo _{i} \neq 0$ \Then $\accIsDelegatedNew _{i} = 1$
			\item \If $\accDelegationAddressNewLo _{i} =    0$ \Then $\accIsDelegatedNew _{i} = 0$
		\end{enumerate}
\end{enumerate}
\saNote{}
The above ensures that
\[
	\left\{ \begin{array}{lclclcl}
		\accIsDelegated    & \equiv & \true & \iff & \accDelegationAddress    & \equiv & \texttt{0x\,00\,\dots\,00} \in \mathbb{B}_{\addressSize} \\
	        \accIsDelegatedNew & \equiv & \true & \iff & \accDelegationAddressNew & \equiv & \texttt{0x\,00\,\dots\,00} \in \mathbb{B}_{\addressSize} \\
	\end{array} \right.
\]
We impose some constraints ensuring that delegation information may only change during the \txAuth{}-phase
\begin{enumerate}[resume]
	\item \If $\txAuth _{i} = 0$ \Then
		\[
			\accSameDelegationAddress {
				anchorRow = i ,
				relOffset = 0 ,
			}
		\]
	\item \If $\accIsDelegated    _{i} = 1$ \Then $\accCodesize     _{i} = \delegatedAccountCodeSize$
	\item \If $\accIsDelegatedNew _{i} = 1$ \Then $\accCodesize\new _{i} = \delegatedAccountCodeSize$
\end{enumerate}
