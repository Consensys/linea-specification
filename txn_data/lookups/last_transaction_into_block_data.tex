		we use the following selector:
		\[
			\col{last\_transaction\_in\_block\_selector} _{i} \define \user_{i} \cdot \sysf_{i+1}
		\]

		\begin{multicols}{2}
			\begin{enumerate}
				\item $\txnDataHubBlockNumber     _{i}$
				\item[\vspace{\fill}]
				\item $\cumulativeConsumedGas     _{i}$
				\item $\cumulativeConsumedBlobGas _{i}$
			\end{enumerate}
		\end{multicols}

		\[
			\col{btc\_target\_selector} _{j} \define \iomf _{j}
		\]

			\end{enumerate}
		\end{multicols}
\end{description}
