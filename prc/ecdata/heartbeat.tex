	\item \locAddressSum{} is \ecDataStamp{}-constant
	\item \If $\ecDataStamp_{i} \neq 0$ \Then we impose
	      \[

	\item \ecdataPhase{} is \ecdataIndex{}-constant \quad (\trash)
\end{enumerate}
